```python
from itertools import chain,combinations
from collections import defaultdict

def solution(relation):
    check = [False]*len(relation[0])
    rel = [i for i in range(len(relation[0]))]
    order = chain.from_iterable(combinations(rel, r) for r in range(1, len(relation[0]) + 1))
    answer = []
    result = 0
    for i in order:
        d = defaultdict(list)
        status = True
        for k in relation:
            flag = ""
            for j in i:
                flag += str(k[j])
            d[flag].append("item")
            if len(d[flag]) > 1:
                status = False
        if status:
            answer.append(i)
    
    for i in answer:
        status = False
        for j in i:
            if check[j]:
                status = True
                break
            else:
                check[j] = True
        if not status:
            result += 1

    return result
```
